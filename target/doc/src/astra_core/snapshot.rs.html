1//! Snapshot manager for deterministic checkpoint creation and restoration.
2//!
3//! Snapshots are hash-verified reconstruction boundaries.
4//! `Snapshot + Journal Replay = Identical State Hash` — that is the invariant.

15const SNAPSHOT_MAGIC: &[u8; 8] = b"ASTRA_SS";
16const SNAPSHOT_VERSION: u32 = 1;

18// =============================================================================
19// SnapshotError
20// =============================================================================
21
22#[derive(Debug)]
23pub enum SnapshotError {
24    IoError(String),
25    InvalidHeader,
26    VersionMismatch { expected: u32, found: u32 },
27    ChecksumMismatch,
28    SerializationError(String),
29    InvalidStateHashLength,
30}
31
32impl fmt::Display for SnapshotError {
33    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
34        match self {
35            Self::IoError(e) => write!(f, "Snapshot I/O error: {}", e),
36            Self::InvalidHeader => write!(f, "Invalid snapshot header"),
37            Self::VersionMismatch { expected, found } => {
38                write!(
39                    f,
40                    "Snapshot version mismatch: expected {}, found {}",
41                    expected, found
42                )
43            }
44            Self::ChecksumMismatch => write!(f, "Snapshot checksum mismatch — data corrupted"),
45            Self::SerializationError(e) => write!(f, "Snapshot serialization error: {}", e),
46            Self::InvalidStateHashLength => write!(f, "State hash must be exactly 32 bytes"),
47        }
48    }
49}
50
51impl std::error::Error for SnapshotError {}
52
53impl From<io::Error> for SnapshotError {
54    fn from(e: io::Error) -> Self {
55        SnapshotError::IoError(e.to_string())
56    }
57}
58
59impl From<SerializationError> for SnapshotError {
60    fn from(e: SerializationError) -> Self {
61        SnapshotError::SerializationError(e.to_string())
62    }
63}

69/// A restored snapshot containing metadata and raw state bytes.
70#[derive(Clone, Debug)]
71pub struct Snapshot {
72    pub metadata: SnapshotMetadata,
73    pub state_data: Vec<u8>,
74    pub created_ns: u64,
75}
76
77impl Snapshot {
78    /// Deserialize snapshot state data into a concrete type.
79    /// Uses the canonical bincode config for determinism.
80    pub fn restore_state<T: serde::de::DeserializeOwned>(&self) -> Result<T, SnapshotError> {
81        deserialize_canonical(&self.state_data)
82            .map_err(|e| SnapshotError::SerializationError(e.to_string()))
83    }
84}

90/// Manages creation and restoration of hash-verified state snapshots.
91pub struct SnapshotManager;
92
93impl SnapshotManager {
94    /// Capture a snapshot from a reducer's current state.
95    ///
96    /// Serializes the reducer state using canonical bincode, computes
97    /// blake3 hash, and writes the snapshot file.
98    pub fn capture<R: EventReducer + serde::Serialize>(
99        reducer: &R,
100        subsystem_id: &str,
101        created_ns: u64,
102        path: &Path,
103    ) -> Result<SnapshotMetadata, SnapshotError> {
104        let state_data = serialize_canonical(reducer)?;
105        let state_hash = hash_bytes(&state_data);
106        let last_seq = reducer.last_applied_sequence_id().unwrap_or(0);
107
108        let mut file = File::create(path)?;
109
110        // Header
111        file.write_all(SNAPSHOT_MAGIC)?;
112        file.write_all(&SNAPSHOT_VERSION.to_le_bytes())?;
113        file.write_all(&created_ns.to_le_bytes())?;
114        file.write_all(&last_seq.to_le_bytes())?;
115        file.write_all(&state_hash)?;
116
117        // Subsystem ID (length-prefixed)
118        let id_bytes = subsystem_id.as_bytes();
119        file.write_all(&(id_bytes.len() as u16).to_le_bytes())?;
120        file.write_all(id_bytes)?;
121
122        // State data (length-prefixed + trailing checksum)
123        file.write_all(&(state_data.len() as u64).to_le_bytes())?;
124        file.write_all(&state_data)?;
125        file.write_all(&state_hash)?; // trailing checksum = same as header hash
126        file.flush()?;
127
128        Ok(SnapshotMetadata::from_hash(
129            last_seq,
130            state_hash,
131            subsystem_id.to_string(),
132        ))
133    }
134
135    /// Restore a snapshot from file. Validates checksum before returning.
136    pub fn restore(path: &Path) -> Result<Snapshot, SnapshotError> {
137        let mut file = File::open(path)?;
138
139        // Read and validate magic
140        let mut magic = [0u8; 8];
141        file.read_exact(&mut magic)?;
142        if &magic != SNAPSHOT_MAGIC {
143            return Err(SnapshotError::InvalidHeader);
144        }
145
146        // Version
147        let mut buf4 = [0u8; 4];
148        file.read_exact(&mut buf4)?;
149        let version = u32::from_le_bytes(buf4);
150        if version != SNAPSHOT_VERSION {
151            return Err(SnapshotError::VersionMismatch {
152                expected: SNAPSHOT_VERSION,
153                found: version,
154            });
155        }
156
157        // Created timestamp
158        let mut buf8 = [0u8; 8];
159        file.read_exact(&mut buf8)?;
160        let created_ns = u64::from_le_bytes(buf8);
161
162        // Last sequence ID
163        file.read_exact(&mut buf8)?;
164        let last_sequence_id = u64::from_le_bytes(buf8);
165
166        // State hash from header
167        let mut header_hash = [0u8; 32];
168        file.read_exact(&mut header_hash)?;
169
170        // Subsystem ID
171        let mut buf2 = [0u8; 2];
172        file.read_exact(&mut buf2)?;
173        let id_len = u16::from_le_bytes(buf2) as usize;
174        let mut id_buf = vec![0u8; id_len];
175        file.read_exact(&mut id_buf)?;
176        let subsystem_id = String::from_utf8_lossy(&id_buf).to_string();
177
178        // State data
179        file.read_exact(&mut buf8)?;
180        let data_len = u64::from_le_bytes(buf8) as usize;
181        let mut state_data = vec![0u8; data_len];
182        file.read_exact(&mut state_data)?;
183
184        // Trailing checksum
185        let mut trailing_checksum = [0u8; 32];
186        file.read_exact(&mut trailing_checksum)?;
187
188        // VALIDATE: compute hash of state_data, compare to both checksums
189        let computed_hash = hash_bytes(&state_data);
190        if computed_hash != header_hash || computed_hash != trailing_checksum {
191            return Err(SnapshotError::ChecksumMismatch);
192        }
193
194        let metadata = SnapshotMetadata::from_hash(last_sequence_id, header_hash, subsystem_id);
195
196        Ok(Snapshot {
197            metadata,
198            state_data,
199            created_ns,
200        })
201    }
202
203    /// Quick-verify a snapshot file's integrity without fully loading it.
204    pub fn verify(path: &Path) -> Result<SnapshotMetadata, SnapshotError> {
205        let snapshot = Self::restore(path)?;
206        Ok(snapshot.metadata)
207    }
208}

221    fn temp_path(name: &str) -> PathBuf {
222        let dir = PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("test_artifacts");
223        fs::create_dir_all(&dir).unwrap();
224        dir.join(name)

227    fn cleanup(path: &Path) {
228        let _ = fs::remove_file(path);
229    }

231    #[derive(Clone, Debug, Serialize, Deserialize, PartialEq, Eq)]
232    struct TestReducer {
233        count: u64,
234        sum: u64,
235        last_seq: Option<u64>,
236    }
237
238    impl TestReducer {
239        fn new() -> Self {
240            Self {
241                count: 0,
242                sum: 0,
243                last_seq: None,
244            }
245        }
246    }
247
248    impl DeterministicState for TestReducer {
249        fn state_hash(&self) -> [u8; 32] {
250            let bytes = serialize_canonical(self).expect("serialization failed");
251            hash_bytes(&bytes)
252        }
253    }
254
255    impl EventReducer for TestReducer {
256        type Error = String;
257        fn apply(&mut self, event: &crate::events::AstraEvent) -> Result<(), String> {
258            self.count += 1;
259            self.sum += event.payload.iter().map(|b| *b as u64).sum::<u64>();
260            self.last_seq = Some(event.sequence_id);
261            Ok(())
262        }
263        fn last_applied_sequence_id(&self) -> Option<u64> {
264            self.last_seq
265        }
266    }
267
268    #[test]
269    fn test_snapshot_capture_and_restore() {
270        let snap_path = temp_path("test_snap.astra_ss");
271        let jl_path = temp_path("test_snap_jl.astra_jl");
272        cleanup(&snap_path);
273        cleanup(&jl_path);
274
275        let mut journal = EventJournal::create(&jl_path, 1_700_000_000_000_000_000).unwrap();
276        let mut reducer = TestReducer::new();
277
278        for i in 1..=5u64 {
279            let event = journal
280                .commit(
281                    1_700_000_000_000_000_000 + i * 1_000_000,
282                    EventType::MarketTick,
283                    vec![i as u8],
284                    PayloadMetadata::raw(),
285                )
286                .unwrap();
287            reducer.apply(&event).unwrap();
288        }
289
290        let meta = SnapshotManager::capture(
291            &reducer,
292            "test-engine",
293            1_700_000_000_000_000_000,
294            &snap_path,
295        )
296        .unwrap();
297        assert_eq!(meta.last_sequence_id, 5);
298        assert_eq!(meta.state_hash, reducer.state_hash());
299
300        let snapshot = SnapshotManager::restore(&snap_path).unwrap();
301        let restored: TestReducer = snapshot.restore_state().unwrap();
302        assert_eq!(restored, reducer);
303        assert_eq!(restored.state_hash(), reducer.state_hash());
304
305        cleanup(&snap_path);
306        cleanup(&jl_path);
307    }
